def find_operating_point(self, r=None):
    """工况点解析法：H0 - kQ² = Hst + rQ²"""
    Q_squared = (self.H0 - self.Hst) / (self.k_pump + r)
    Q = np.sqrt(Q_squared)
    H = self.pump_curve(Q)
    return Q, H
