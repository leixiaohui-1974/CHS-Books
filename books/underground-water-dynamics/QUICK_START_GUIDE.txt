📌 第二步：安装环境
─────────────────────────────────────────────────────────
  # 创建虚拟环境
  conda create -n gwflow python=3.9
  conda activate gwflow
  
  # 安装依赖
  pip install -r requirements.txt
  
  # 安装gwflow包（开发模式）
  pip install -e .

📌 第三步：运行测试
─────────────────────────────────────────────────────────
  # 简单测试（不需要pytest）
  python3 tests/run_simple_tests.py
  
  # 完整测试（需要pytest）
  pytest tests/ -v

📌 第四步：运行案例
─────────────────────────────────────────────────────────

  python3 case_01_1d_steady.py
  
  # 案例2：二维稳态
  cd ../case_02

  # 案例4：瞬态
