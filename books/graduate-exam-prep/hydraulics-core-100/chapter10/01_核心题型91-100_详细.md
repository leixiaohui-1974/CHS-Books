from scipy.optimize import fsolve

class OpenChannelDesign:
    """æ¸ é“ä¼˜åŒ–è®¾è®¡"""
    
    def __init__(self, Q, i, n, m, g=9.81):
        """
        å‚æ•°:
            Q: è®¾è®¡æµé‡(mÂ³/s)
            i: åº•å¡
            n: ç³™ç‡
            m: è¾¹å¡ç³»æ•°
            g: é‡åŠ›åŠ é€Ÿåº¦(m/sÂ²)
        """
        self.Q = Q
        self.i = i
        self.n = n
        self.m = m
        self.g = g
    
    def optimal_trapezoid(self):
        """
        æ°´åŠ›æœ€ä¼˜æ¢¯å½¢æ–­é¢
        
        è¿”å›:
            h: æ°´æ·±(m)
            b: åº•å®½(m)

        """
        # æ°´åŠ›æœ€ä¼˜æ¡ä»¶
        b_h_ratio = 2 * (np.sqrt(1 + self.m**2) - self.m)
        
        def equation(h):
            b = b_h_ratio * h
            A = (b + self.m * h) * h
            chi = b + 2 * h * np.sqrt(1 + self.m**2)
            R = A / chi
            
            Q_calc = (1/self.n) * A * R**(2/3) * np.sqrt(self.i)
            return Q_calc - self.Q
        
        # æ±‚è§£
        h_solution = fsolve(equation, 1.0)[0]
        b_solution = b_h_ratio * h_solution
        
        A = (b_solution + self.m * h_solution) * h_solution
        v = self.Q / A
        
        return h_solution, b_solution, v
    
    def design_with_velocity_constraint(self, v_min=0.6, v_max=1.2):
        """
        è€ƒè™‘æµé€Ÿçº¦æŸçš„è®¾è®¡

            v_min, v_max: å…è®¸æµé€ŸèŒƒå›´(m/s)
        """
        h_opt, b_opt, v_opt = self.optimal_trapezoid()
        
        print(f"æ°´åŠ›æœ€ä¼˜æ–­é¢:")
        print(f"  h = {h_opt:.3f} m, b = {b_opt:.3f} m, v = {v_opt:.2f} m/s")
        
        if v_min <= v_opt <= v_max:
            print(f"  âœ“ æ»¡è¶³æµé€Ÿçº¦æŸ")
            return h_opt, b_opt, v_opt
        elif v_opt > v_max:
            print(f"  âœ— æµé€Ÿè¿‡å¤§ï¼Œéœ€å¢å¤§æ–­é¢")
            # æŒ‰æœ€å¤§æµé€Ÿè®¾è®¡
            A_required = self.Q / v_max
            
            def equation_A(h):
                b = 2 * (np.sqrt(1 + self.m**2) - self.m) * h
                A = (b + self.m * h) * h
                return A - A_required
            
            h_new = fsolve(equation_A, h_opt)[0]
            b_new = 2 * (np.sqrt(1 + self.m**2) - self.m) * h_new
            v_new = self.Q / A_required
            
            print(f"\nä¿®æ­£æ–­é¢:")
            print(f"  h = {h_new:.3f} m, b = {b_new:.3f} m, v = {v_new:.2f} m/s")
            
            return h_new, b_new, v_new
        else:
            print(f"  âœ— æµé€Ÿè¿‡å°ï¼Œéœ€å‡å°æ–­é¢")
            return h_opt, b_opt, v_opt
