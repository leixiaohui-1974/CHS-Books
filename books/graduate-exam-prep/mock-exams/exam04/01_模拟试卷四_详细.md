A. 380 mm  
B. 543 mm  
C. 620 mm  
D. 714 mm

**【答案】B**

**【解析】**
$$m_{净} = 500 - 120 = 380 \text{ mm}$$
$$m_{毛} = \frac{380}{0.70} = 543 \text{ mm}$$

### 计算题1：土坝渗流计算（20分）

**题目**：

均质土坝，坝高H=40m，上游水深H₁=36m，下游水深H₂=4m，渗透系数k=2×10⁻⁵m/s，渗径长度L=120m。

求：
(1) 单宽渗流量  
(2) 若坝长500m，总渗流量  
(3) 出溢坡降（出溢点水深5m）

**【解答】**

**(1) 单宽渗流量**（8分）

$$q = \frac{k(H_1^2 - H_2^2)}{2L}$$

$$q = \frac{2 \times 10^{-5} \times (36^2 - 4^2)}{2 \times 120}$$

$$q = \frac{2 \times 10^{-5} \times 1280}{240} = 1.07 \times 10^{-4} \text{ m}^2\text{/s}$$

**(2) 总渗流量**（6分）

$$Q = q \times L_{坝} = 1.07 \times 10^{-4} \times 500 = 0.053 \text{ m}^3\text{/s}$$

$$Q = 0.053 \times 86400 = 4579 \text{ m}^3\text{/d} \approx 4600 \text{ m}^3\text{/d}$$

**(3) 出溢坡降**（6分）

$$i_{出} = \frac{q}{k \times h_{出}} = \frac{1.07 \times 10^{-4}}{2 \times 10^{-5} \times 5} = 1.07$$

**评分点**：
- 公式正确（3分）
- 代入计算（2分）
- 结果与单位（1分）

import numpy as np

class ExamProblemSolver:
    """考试题目求解器"""
    
    @staticmethod
    def earth_dam_seepage(k, H1, H2, L, L_dam=None, h_exit=None):
        """
        土坝渗流计算
        
        参数:
            k: 渗透系数 (m/s)
            H1: 上游水深 (m)
            H2: 下游水深 (m)
            L: 渗径长度 (m)
            L_dam: 坝长 (m)
            h_exit: 出溢点水深 (m)
        """
        # (1) 单宽渗流量
        q = k * (H1**2 - H2**2) / (2 * L)
        
        print(f"土坝渗流计算:")
        print(f"  渗透系数 k = {k} m/s")
        print(f"  上游水深 H₁ = {H1} m")
        print(f"  下游水深 H₂ = {H2} m")
        print(f"  渗径长度 L = {L} m")
        
        print(f"\n(1) 单宽渗流量:")
        print(f"  q = {q:.2e} m²/s = {q*86400:.4f} m²/d")
        
        # (2) 总渗流量
        if L_dam:
            Q = q * L_dam
            print(f"\n(2) 总渗流量（坝长{L_dam}m）:")
            print(f"  Q = {Q:.4f} m³/s = {Q*86400:.0f} m³/d")
        
        # (3) 出溢坡降
        if h_exit:
            i_exit = q / (k * h_exit)
            print(f"\n(3) 出溢坡降:")
            print(f"  i_出 = {i_exit:.3f}")
            
            # 临界坡降判断
            i_cr = 1.0  # 典型值
            if i_exit < i_cr:
                print(f"  ✓ 渗透稳定（i_出 < i_cr={i_cr}）")
            else:
                print(f"  ✗ 渗透不稳定（i_出 ≥ i_cr={i_cr}）")
    
    @staticmethod
    def irrigation_calculation(ET_c, P_eff, eta, A):
        """
        灌溉计算
        
        参数:
            ET_c: 作物腾发量 (mm)
            P_eff: 有效降雨 (mm)
            eta: 灌溉水利用系数
            A: 灌溉面积 (亩)
        """
        # 净灌溉定额
        m_net = max(0, ET_c - P_eff)
        
        # 毛灌溉定额
        m_gross = m_net / eta
        
        # 总供水量（1亩=666.67m²）
        V_total = m_gross / 1000 * A * 666.67
        
        print(f"灌溉计算:")
        print(f"  作物腾发量 ET_c = {ET_c} mm")
        print(f"  有效降雨 P_eff = {P_eff} mm")
        print(f"  灌溉水利用系数 η = {eta}")
        print(f"  灌溉面积 A = {A} 亩")
        
        print(f"\n结果:")
        print(f"  净灌溉定额 m_净 = {m_net:.1f} mm")
        print(f"  毛灌溉定额 m_毛 = {m_gross:.1f} mm")
        print(f"  总供水量 V = {V_total:.0f} m³ = {V_total/10000:.2f} 万m³")
    
    @staticmethod
    def water_hammer_calculation(L, v0, a):
        """
        水锤计算
        
        参数:
            L: 管长 (m)
            v0: 初始流速 (m/s)
            a: 波速 (m/s)
        """
        g = 9.81
        
        # 直接水锤压强升高
        delta_H = a * v0 / g
        
        # 临界关闭时间
        T_c = 2 * L / a
        
        print(f"水锤计算:")
        print(f"  管长 L = {L} m")
        print(f"  初始流速 v₀ = {v0} m/s")
        print(f"  波速 a = {a} m/s")
        
        print(f"\n结果:")
        print(f"  直接水锤压强升高 ΔH = {delta_H:.1f} m")
        print(f"  临界关闭时间 T_c = 2L/a = {T_c:.2f} s")
        
        # 防护建议
        if delta_H > 50:
            print(f"\n防护建议:")
            print(f"  - 压强升高较大（>{delta_H:.0f}m），建议设置:")
            print(f"    1. 调压塔/安全阀")
            print(f"    2. 空气罐")
            print(f"    3. 缓慢关阀（T_c > {T_c:.1f}s）")
