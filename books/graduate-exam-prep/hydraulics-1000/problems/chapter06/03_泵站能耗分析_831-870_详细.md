## 题831：泵站年能耗计算

**【题目】**（30分，考频⭐⭐⭐⭐⭐）

某灌溉泵站，设计参数：
- 装机流量：Q = 3.0 m³/s
- 设计扬程：H = 40 m
- 水泵效率：η_p = 82%
- 电机效率：η_m = 95%
- 年运行时间：T = 2000 h
- 电价：0.55 元/kWh

求：
1. 轴功率
2. 电机输入功率
3. 年耗电量
4. 年电费
5. 单位水量能耗

**【解答】**

### 1. 轴功率

$$P_{轴} = \frac{\rho g Q H}{\eta_p} = \frac{1000 \times 9.8 \times 3.0 \times 40}{0.82}$$
$$= 1,434,146 \text{ W} = 1434.1 \text{ kW}$$

### 2. 电机输入功率

$$P_{电} = \frac{P_{轴}}{\eta_m} = \frac{1434.1}{0.95} = 1509.6 \text{ kW}$$

### 3. 年耗电量

$$W_{年} = P_{电} \times T = 1509.6 \times 2000 = 3,019,200 \text{ kWh}$$
$$= 301.92 \text{ 万kWh}$$

### 4. 年电费

$$C_{年} = W_{年} \times 电价 = 3,019,200 \times 0.55$$
$$= 1,660,560 \text{ 元} = 166.06 \text{ 万元}$$

### 5. 单位水量能耗

总提水量：
$$V_{年} = Q \times T \times 3600 = 3.0 \times 2000 \times 3600$$
$$= 21,600,000 \text{ m³} = 2160 \text{ 万m³}$$

单位能耗：
$$e = \frac{W_{年}}{V_{年}} = \frac{3,019,200}{21,600,000} = 0.1398 \text{ kWh/m³}$$

### Python实现

```python
import numpy as np
import matplotlib.pyplot as plt
import pandas as pd

class PumpStationEnergyAnalysis:
    """
    泵站能耗分析
    """
    
    def __init__(self, Q, H, eta_p=0.82, eta_m=0.95, rho=1000, g=9.8):
        """
        参数:
            Q: 流量 (m³/s)
            H: 扬程 (m)
            eta_p: 水泵效率
            eta_m: 电机效率
            rho: 水密度 (kg/m³)
            g: 重力加速度 (m/s²)
        """
        self.Q = Q
        self.H = H
        self.eta_p = eta_p
        self.eta_m = eta_m
        self.rho = rho
        self.g = g
    
    def shaft_power(self):
        """
        轴功率 (kW)
        """
        P = (self.rho * self.g * self.Q * self.H) / (self.eta_p * 1000)
        return P
    
    def input_power(self):
        """
        电机输入功率 (kW)
        """
        P_shaft = self.shaft_power()
        P_input = P_shaft / self.eta_m
        return P_input
    
    def annual_energy(self, T_hours):
        """
        年耗电量
        
        参数:
            T_hours: 年运行时间 (h)
        
        返回:
            年耗电量 (kWh), 年提水量 (m³)
        """
        P_input = self.input_power()
        
        W_annual = P_input * T_hours  # kWh
        V_annual = self.Q * T_hours * 3600  # m³
        
        return W_annual, V_annual
    
    def specific_energy(self, T_hours):
        """
        单位水量能耗
        
        返回:
            kWh/m³
        """
        W, V = self.annual_energy(T_hours)
        e = W / V
        return e
    
    def annual_cost(self, T_hours, electricity_price):
        """
        年运行费用
        
        参数:
            T_hours: 年运行时间 (h)
            electricity_price: 电价 (元/kWh)
        
        返回:
            年电费 (元)
        """
        W_annual, _ = self.annual_energy(T_hours)
        cost = W_annual * electricity_price
        return cost
    
    def energy_efficiency_ratio(self):
        """
        能效比（综合效率）
        
        η_综 = η_p × η_m
        """
        eta_total = self.eta_p * self.eta_m
        return eta_total
    
    def sensitivity_analysis(self, T_hours, electricity_price, 
                            param_name, param_range):
        """
        敏感性分析
        
        参数:
            T_hours: 运行时间
            electricity_price: 电价
            param_name: 参数名 ('eta_p', 'eta_m', 'H')
            param_range: 参数变化范围
        
        返回:
            参数值列表, 成本列表
        """
        costs = []
        
        for param_val in param_range:
            # 临时修改参数
            if param_name == 'eta_p':
                temp = PumpStationEnergyAnalysis(
                    self.Q, self.H, param_val, self.eta_m
                )
            elif param_name == 'eta_m':
                temp = PumpStationEnergyAnalysis(
                    self.Q, self.H, self.eta_p, param_val
                )
            elif param_name == 'H':
                temp = PumpStationEnergyAnalysis(
                    self.Q, param_val, self.eta_p, self.eta_m
                )
            else:
                continue
            
            cost = temp.annual_cost(T_hours, electricity_price)
            costs.append(cost / 10000)  # 万元
        
        return param_range, costs
    
    def plot_energy_breakdown(self, T_hours, electricity_price):
        """
        绘制能耗分解图
        """
        # 计算各项功率和损失
        P_useful = (self.rho * self.g * self.Q * self.H) / 1000  # kW
        P_pump_loss = P_useful / self.eta_p - P_useful
        P_motor_loss = (P_useful + P_pump_loss) / self.eta_m - (P_useful + P_pump_loss)
        P_total = P_useful + P_pump_loss + P_motor_loss
        
        # 年能耗
        W_useful = P_useful * T_hours / 10000  # 万kWh
        W_pump_loss = P_pump_loss * T_hours / 10000
        W_motor_loss = P_motor_loss * T_hours / 10000
        
        fig, axes = plt.subplots(1, 2, figsize=(14, 6))
        
        # 功率分解
        labels1 = ['有效功率', '水泵损失', '电机损失']
        sizes1 = [P_useful, P_pump_loss, P_motor_loss]
        colors1 = ['#2ecc71', '#f39c12', '#e74c3c']
        explode1 = (0.05, 0, 0)
        
        axes[0].pie(sizes1, explode=explode1, labels=labels1, 
                   colors=colors1, autopct='%1.1f%%',
                   shadow=True, startangle=90)
        axes[0].set_title(f'功率分解\n总功率={P_total:.1f}kW', fontsize=14)
        
        # 年能耗成本
        costs = [W * electricity_price * 10000 / 10000 for W in 
                [W_useful, W_pump_loss, W_motor_loss]]
        
        bars = axes[1].bar(labels1, costs, color=colors1, alpha=0.7)
        axes[1].set_ylabel('年成本 (万元)', fontsize=12)
        axes[1].set_title(f'年能耗成本\n总计={sum(costs):.1f}万元', fontsize=14)
        axes[1].grid(True, alpha=0.3, axis='y')
        
        # 标注数值
        for bar in bars:
            height = bar.get_height()
            axes[1].text(bar.get_x() + bar.get_width()/2., height,
                        f'{height:.1f}',
                        ha='center', va='bottom', fontsize=11)
        
        plt.tight_layout()
        plt.savefig('energy_breakdown.png', dpi=300, bbox_inches='tight')
        plt.show()


# 使用示例
print("="*60)
print("泵站能耗分析")
print("="*60)

# 创建分析对象

    Q=3.0, H=40, eta_p=0.82, eta_m=0.95
)

# 基本计算

print(f"功率计算:")

# 年运行参数

W_annual, V_annual = pump_station.annual_energy(T_hours)

print(f"\n年运行分析:")
print(f"  运行时间: {T_hours} h")
print(f"  年耗电量: {W_annual/10000:.2f} 万kWh")
print(f"  年提水量: {V_annual/10000:.2f} 万m³")
print(f"  单位能耗: {e_specific:.4f} kWh/m³")
print(f"  年电费: {cost_annual/10000:.2f} 万元")

# 能效比

print(f"\n综合效率: {eta_total*100:.2f}%")
print(f"  水泵效率: {pump_station.eta_p*100:.0f}%")
print(f"  电机效率: {pump_station.eta_m*100:.0f}%")

# 能耗分解
pump_station.plot_energy_breakdown(T_hours, electricity_price)

# 敏感性分析
print("\n" + "="*60)
print("敏感性分析")
print("="*60)

fig, axes = plt.subplots(1, 3, figsize=(16, 5))

# 1. 水泵效率影响
eta_p_range = np.linspace(0.70, 0.90, 20)

    T_hours, electricity_price, 'eta_p', eta_p_range
)

axes[0].plot(eta_p_vals*100, costs_p, 'b-o', linewidth=2, markersize=4)
axes[0].axvline(pump_station.eta_p*100, color='r', linestyle='--',
               label=f'当前值={pump_station.eta_p*100:.0f}%')
axes[0].set_xlabel('水泵效率 (%)', fontsize=12)
axes[0].set_ylabel('年电费 (万元)', fontsize=12)
axes[0].set_title('水泵效率影响', fontsize=14)
axes[0].legend()
axes[0].grid(True, alpha=0.3)

# 2. 电机效率影响
eta_m_range = np.linspace(0.88, 0.98, 20)
eta_m_vals, costs_m = pump_station.sensitivity_analysis(
    T_hours, electricity_price, 'eta_m', eta_m_range
)

axes[1].plot(eta_m_vals*100, costs_m, 'g-o', linewidth=2, markersize=4)
axes[1].axvline(pump_station.eta_m*100, color='r', linestyle='--',
               label=f'当前值={pump_station.eta_m*100:.0f}%')
axes[1].set_xlabel('电机效率 (%)', fontsize=12)
axes[1].set_ylabel('年电费 (万元)', fontsize=12)
axes[1].set_title('电机效率影响', fontsize=14)
axes[1].legend()
axes[1].grid(True, alpha=0.3)

# 3. 扬程影响
H_range = np.linspace(30, 50, 20)
H_vals, costs_H = pump_station.sensitivity_analysis(
    T_hours, electricity_price, 'H', H_range
)

axes[2].plot(H_vals, costs_H, 'orange', marker='o', 
            linewidth=2, markersize=4)
axes[2].axvline(pump_station.H, color='r', linestyle='--',
               label=f'当前值={pump_station.H:.0f}m')
axes[2].set_xlabel('扬程 (m)', fontsize=12)
axes[2].set_ylabel('年电费 (万元)', fontsize=12)
axes[2].set_title('扬程影响', fontsize=14)
```

## 题832：节能潜力分析

**【题目】**（25分，考频⭐⭐⭐⭐）

上题泵站，若通过技术改造：
1. 水泵效率提高到85%
2. 采用变频调速，综合节电15%

计算改造后的节能效果和投资回收期（改造费用50万元）。

**【解答】**

### 1. 效率提升节能

原年电费：166.06万元

提效后电机功率：
$$P_{新} = \frac{1434.1 \times \frac{0.82}{0.85}}{0.95} = 1455.4 \text{ kW}$$

年电费：
$$C_{新1} = 1455.4 \times 2000 \times 0.55 / 10000 = 160.09 \text{ 万元}$$

节省：$166.06 - 160.09 = 5.97$ 万元/年

### 2. 变频调速节能

综合节电15%：
$$C_{新2} = 160.09 \times (1 - 0.15) = 136.08 \text{ 万元}$$

总节省：$166.06 - 136.08 = 29.98$ 万元/年

### 3. 投资回收期

$$T = \frac{投资}{年节约} = \frac{50}{29.98} = 1.67 \text{ 年}$$

**结论**：投资回收期短，经济效益显著！
