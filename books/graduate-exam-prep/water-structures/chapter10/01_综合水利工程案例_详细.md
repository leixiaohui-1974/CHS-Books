## 一、三峡水利枢纽综合案例

### 1.1 工程概况

**基本参数**：
- 坝址：湖北省宜昌市三斗坪
- 坝型：混凝土重力坝
- 坝顶高程：185 m
- 正常蓄水位：175 m
- 防洪限制水位：145 m（汛期）
- 总库容：393亿m³
- 防洪库容：221.5亿m³
- 装机容量：22,500 MW
- 年发电量：约1000亿kW·h

**工程任务**：
1. 防洪
2. 发电
3. 航运
4. 供水
5. 生态调度

---

### 1.2 防洪调度分析

**防洪目标**：
- 荆江河段防洪标准：100年一遇
- 城陵矶地区防洪标准：提高到10-20年一遇

**调度原则**：
1. 汛前降至防洪限制水位145m
2. 汛期拦蓄超额洪水
3. 汛末腾库至145m
4. 非汛期蓄水至175m

**防洪计算**：

```python
import numpy as np
import matplotlib.pyplot as plt
from scipy.interpolate import interp1d


class ThreeGorgesReservoir:
    """三峡水库综合调度模拟"""
    
    def __init__(self):
        # 水位-库容关系（简化）
        self.Z_levels = np.array([145, 150, 155, 160, 165, 170, 175])  # m
        self.V_storage = np.array([171.3, 193.9, 218.0, 244.2, 273.1, 
                                   305.7, 393.0]) * 1e8  # m^3
        
        self.Z_V_func = interp1d(self.Z_levels, self.V_storage, 
                                kind='cubic', fill_value='extrapolate')
        self.V_Z_func = interp1d(self.V_storage, self.Z_levels,
                                kind='cubic', fill_value='extrapolate')
        
        # 下泄流量-水位关系（简化，不考虑泥沙）
        self.Q_max = 110000  # m^3/s（最大下泄能力）
        
        # 发电参数
        self.H_tail = 62  # 尾水位 (m)
        self.eta = 0.90  # 综合效率
        self.P_installed = 22500  # MW
    
    def flood_routing(self, Q_in_series, dt=3600, Z_init=145):
        """
        洪水调节计算
        
        参数:
            Q_in_series: 入库流量过程 (m^3/s)
            dt: 时间步长 (s)
            Z_init: 初始水位 (m)
        
        返回:
            Z_series: 水位过程
            Q_out_series: 出库流量过程
            V_series: 库容过程
        """
        n = len(Q_in_series)
        
        Z_series = np.zeros(n)
        Q_out_series = np.zeros(n)
        V_series = np.zeros(n)
        
        # 初始化
        Z_series[0] = Z_init
        V_series[0] = self.Z_V_func(Z_init)
        
        for i in range(n - 1):
            Q_in = Q_in_series[i]
            V_current = V_series[i]
            Z_current = Z_series[i]
            
            # 调度规则
            if Z_current < 145:
                # 低于防洪限制水位：尽量少泄
                Q_out = min(Q_in, 20000)
            elif Z_current < 155:
                # 145-155m：控制下泄
                Q_out = min(Q_in, 30000)
            elif Z_current < 165:
                # 155-165m：加大下泄
                Q_out = min(Q_in * 0.8, 50000)
            elif Z_current < 175:
                # 165-175m：大量下泄
                Q_out = min(Q_in * 0.9, 80000)
            else:
                # 超过175m：全力下泄
                Q_out = min(Q_in, self.Q_max)
            
            # 水量平衡
            V_next = V_current + (Q_in - Q_out) * dt
            
            # 边界处理
            V_min = self.Z_V_func(145)
            V_max = self.Z_V_func(175)
            
            if V_next < V_min:
                V_next = V_min
                Q_out = Q_in - (V_next - V_current) / dt
            elif V_next > V_max:
                V_next = V_max
                Q_out = Q_in - (V_next - V_current) / dt
            
            V_series[i+1] = V_next
            Z_series[i+1] = self.V_Z_func(V_next)
            Q_out_series[i] = Q_out
        
        Q_out_series[-1] = Q_out_series[-2]
        
        return Z_series, Q_out_series, V_series
    
    def power_generation(self, Q, Z):
        """
        发电量计算
        
        参数:
            Q: 发电流量 (m^3/s)
            Z: 上游水位 (m)
        
        返回:
            P: 发电量 (MW)
        """
        H = Z - self.H_tail
        P = 9.81 * self.eta * Q * H / 1000  # MW
        
        # 不超过装机容量
        P = min(P, self.P_installed)
        
        return P
    
    def annual_operation(self, monthly_inflow):
        """
        年度优化调度
        
        参数:
            monthly_inflow: 12个月平均入流 (m^3/s)
        
        返回:
            operation_plan: 调度方案
        """
        months = ['1月', '2月', '3月', '4月', '5月', '6月',
                 '7月', '8月', '9月', '10月', '11月', '12月']
        
        Z_plan = np.zeros(13)  # 13个时刻（包括年初年末）
        
        # 汛期（6-9月）：145m
        # 汛末（10月）：逐步蓄水
        # 非汛期（11-次年5月）：175m
        Z_plan[0] = 175  # 1月初
        Z_plan[1:6] = 175  # 1-5月
        Z_plan[6:10] = 145  # 6-9月
        Z_plan[10] = 160  # 10月（蓄水中）
        Z_plan[11:13] = 175  # 11-12月
        
        Q_gen = []
        P_gen = []
        
        for i in range(12):
            Q_avg = monthly_inflow[i]
            Z_avg = (Z_plan[i] + Z_plan[i+1]) / 2
            
            # 发电流量（考虑弃水）
            if Q_avg > 60000:
                Q_g = 60000  # 机组最大过流能力
            else:
                Q_g = Q_avg
            
            P = self.power_generation(Q_g, Z_avg)
            
            Q_gen.append(Q_g)
            P_gen.append(P)
        
        return {
            'months': months,
            'Z_plan': Z_plan,
            'Q_gen': Q_gen,
            'P_gen': P_gen,
            'E_annual': sum(P_gen) * 30 * 24 / 1e6  # TWh
        }
```

### 2.2 长距离输水计算

```python
class LongDistanceWaterTransfer:
    """长距离输水工程分析"""
    
    def __init__(self, L_total, n_sections, Q_design):
        """
        参数:
            L_total: 总长度 (km)
            n_sections: 分段数
            Q_design: 设计流量 (m^3/s)
        """
        self.L_total = L_total * 1000  # 转为m
        self.n_sections = n_sections
        self.Q_design = Q_design
        
        self.dx = self.L_total / n_sections
        self.x = np.linspace(0, self.L_total, n_sections + 1)
    
    def headloss_channel(self, b, h, m, n, S0, L):
        """
        渠道水头损失
        
        参数:
            b: 底宽 (m)
            h: 水深 (m)
            m: 边坡系数
            n: 糙率
            S0: 底坡
            L: 长度 (m)
        
        返回:
            h_f: 沿程损失 (m)
        """
        A = (b + m * h) * h
        P = b + 2 * h * np.sqrt(1 + m**2)
        R = A / P
        
        v = self.Q_design / A
        
        # Manning公式
        S_f = (n * v / R**(2/3))**2
        
        h_f = S_f * L
        
        return h_f
    
    def aqueduct_design(self, L_aq, H_terrain, b_aq=8):
        """
        渡槽设计
        
        参数:
            L_aq: 渡槽长度 (m)
            H_terrain: 地形高差 (m)
            b_aq: 渡槽宽度 (m)
        
        返回:
            design: 设计参数
        """
        # 渡槽水深（按均匀流）
        n_aq = 0.014  # 混凝土
        S0_aq = 0.0001  # 底坡
        
        # 迭代求水深
        h_aq = 2.0  # 初值
        for _ in range(10):
            A = b_aq * h_aq
            P = b_aq + 2 * h_aq
            R = A / P
            
            h_new = (self.Q_design * n_aq / (b_aq * np.sqrt(S0_aq)))**(3/5)
            
            if abs(h_new - h_aq) < 0.01:
                break
            h_aq = h_new
        
        # 墩柱高度
        H_pier = H_terrain + 5  # 安全余量
        
        # 单孔跨度（优化，不超过60m）
        L_span = min(60, L_aq / 20)
        n_spans = int(np.ceil(L_aq / L_span))
        
        design = {
            'L_aq': L_aq,
            'b_aq': b_aq,
            'h_aq': h_aq,
            'H_pier': H_pier,
            'L_span': L_span,
            'n_spans': n_spans,
            'V_concrete': b_aq * h_aq * L_aq * 0.2  # 简化体积估算
        }
        
        return design
    
    def pumping_station_design(self, H_lift, eta_pump=0.85):
        """
        泵站设计
        
        参数:
            H_lift: 扬程 (m)
            eta_pump: 泵效率
        
        返回:
            P_installed: 装机功率 (MW)
            E_annual: 年耗电 (GW·h)
        """
        g = 9.81
        rho = 1000
        
        # 单泵功率
        P_pump = rho * g * self.Q_design * H_lift / (eta_pump * 1e6)  # MW
        
        # 装机（3台，1备）
        P_installed = P_pump * 3
        
        # 年运行时间（按调水计划）
        T_annual = 300 * 24  # 小时/年
        
        # 年耗电
        E_annual = P_pump * 2 * T_annual / 1000  # GW·h
        
        return P_installed, E_annual
    
    def economic_analysis(self, cost_channel=1000, cost_aqueduct=5000,
                         cost_tunnel=10000, price_water=3.0):
        """
        经济分析
        
        参数:
            cost_channel: 渠道单价 (元/m³土方)
            cost_aqueduct: 渡槽单价 (元/m³混凝土)
            cost_tunnel: 隧洞单价 (元/m³开挖)
            price_water: 水价 (元/m³)
        
        返回:
            analysis: 经济指标
        """
        # 假设工程组成
        L_channel = self.L_total * 0.6
        L_aqueduct = self.L_total * 0.1
        L_tunnel = self.L_total * 0.3
        
        # 工程量（简化）
        V_channel = 5 * 3 * L_channel  # m^3
        V_aqueduct = 8 * 2 * L_aqueduct * 0.2  # m^3
        V_tunnel = np.pi * 7**2 / 4 * L_tunnel  # m^3
        
        # 建设投资
        Cost_channel = V_channel * cost_channel
        Cost_aqueduct = V_aqueduct * cost_aqueduct
        Cost_tunnel = V_tunnel * cost_tunnel
        Cost_total = Cost_channel + Cost_aqueduct + Cost_tunnel
        
        # 年调水量（亿m³）
        V_annual = self.Q_design * 300 * 24 * 3600 / 1e8
        
        # 年收益
        Revenue_annual = V_annual * 1e8 * price_water
        
        # 静态投资回收期
        Payback = Cost_total / Revenue_annual
        
        analysis = {
            'Cost_total': Cost_total / 1e8,  # 亿元
            'V_annual': V_annual,  # 亿m³
            'Revenue_annual': Revenue_annual / 1e8,  # 亿元/年
            'Payback': Payback  # 年
        }
        
        return analysis
```

### 3.2 泥沙问题综合分析

```python
class XiaolangdiReservoir:
    """小浪底水库泥沙调度"""
    
    def __init__(self, V_total=126.5e8, V_dead=75.5e8):
        """
        参数:
            V_total: 总库容 (m^3)
            V_dead: 死库容 (m^3)
        """
        self.V_total = V_total
        self.V_dead = V_dead
        self.V_active = V_total - V_dead
    
    def sediment_balance(self, Q_w, S_in, eta_排沙=0.6):
        """
        泥沙平衡计算
        
        参数:
            Q_w: 水流量 (m^3/s)
            S_in: 入库含沙量 (kg/m^3)
            eta_排沙: 排沙效率
        
        返回:
            S_out: 出库含沙量
            W_deposit: 淤积量 (t/s)
        """
        # 入库泥沙量
        W_in = Q_w * S_in  # kg/s
        
        # 排出泥沙量
        W_out = W_in * eta_排沙
        
        # 淤积量
        W_deposit = W_in - W_out
        
        S_out = W_out / Q_w if Q_w > 0 else 0
        
        return S_out, W_deposit / 1000  # t/s
    
    def flushing_operation(self, Q_flush, duration):
        """
        调水调沙运用
        
        参数:
            Q_flush: 冲刷流量 (m^3/s)
            duration: 持续时间 (天)
        
        返回:
            V_flush: 冲刷水量 (亿m³)
            W_flush: 冲出泥沙 (亿t)
        """
        # 冲刷水量
        V_flush = Q_flush * duration * 86400 / 1e8  # 亿m³
        
        # 冲出泥沙（经验公式，简化）
        S_avg = 50  # kg/m³（高含沙水流）
        W_flush = V_flush * 1e8 * S_avg / 1e12  # 亿t
        
        return V_flush, W_flush
    
    def long_term_simulation(self, years=20, W_annual=16e8):
        """
        长期淤积模拟
        
        参数:
            years: 模拟年数
            W_annual: 年均来沙量 (t)
        
        返回:
            V_loss_series: 库容损失序列
        """
        V_loss_series = np.zeros(years + 1)
        
        # 简化：线性淤积模型
        for year in range(1, years + 1):
            # 淤积体积（干容重1.4 t/m³）
            W_deposit = W_annual * 0.5  # 50%淤积
            V_deposit = W_deposit / 1.4
            
            V_loss_series[year] = V_loss_series[year-1] + V_deposit
        
        return V_loss_series / 1e8  # 亿m³
```
